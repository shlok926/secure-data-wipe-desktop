<body>

  <script>
    $(document).ready(function () {
      // Ask user for ages
      let myAge = parseInt(prompt("Enter your age:"));
      let brotherAge = parseInt(prompt("Enter your brother's age:"));

      // Store ages and check eligibility using a Boolean variable
      // Ages must be non-negative (> 0) and both must be less than 25
      let isEligible = myAge > 0 && brotherAge > 0 && myAge < 25 && brotherAge < 25;

      // Print result to the page using DOM/jQuery
      $("body").append("<p>" + isEligible + "</p>");
    });
  </script>
